\begin{definition}
Define the function $g$ by the formula $g(k) = -3|4k-5| + 1$ with domain $(-\infty, \infty)$.

\end{definition}



\begin{exercise}
The graph of $g(k)$ is best described by which of the following?

\begin{multipleChoice}
\choice{Line}
\choice{Ray}
\choice[correct]{V-shaped}
\choice{Parabola}
\choice{Sideways Half-Parabola}
\end{multipleChoice}

\end{exercise}




\begin{exercise}
The graph of $g(k)$ includes what type of point?

\begin{multipleChoice}
\choice[correct]{a highest point}
\choice{a lowest point}
\choice{neither}
\end{multipleChoice}

\end{exercise}



\begin{exercise}
The graph of $g(k)$ opens up or down?

\begin{multipleChoice}
\choice{up}
\choice[correct]{down}
\choice{neither}
\end{multipleChoice}

\end{exercise}





\begin{exercise}
Which domain number is associated with this extreme point?
domain number is $\answer{5/4}$.

\end{exercise}









\begin{exercise}
How many $k$-intercepts does the graph of $g(k)$ have?

\begin{multipleChoice}
\choice{$0$}
\choice{$1$}
\choice[correct]{$2$}
\end{multipleChoice}

\end{exercise}
