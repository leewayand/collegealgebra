\begin{definition}
Define the function $h$ by the formula $h(n) = -2(3n-5)^2+3$ with domain $(-\infty, \infty)$.

\end{definition}



\begin{exercise}
The graph of $h(n)$ is best described by which of the following?

\begin{multipleChoice}
\choice{$Line$}
\choice{$V-shaped$}
\choice[correct]{$Parabola$}
\choice{$Sideways Parabola$}
\end{multipleChoice}

\end{exercise}




\begin{exercise}
The graph of $h(n)$ includes what type of point?

\begin{multipleChoice}
\choice[correct]{$a highest point$}
\choice{$a lowest point$}
\choice{$neither$}
\end{multipleChoice}

\end{exercise}



\begin{exercise}
The graph of $h(n)$ opens up or down?

\begin{multipleChoice}
\choice{$up$}
\choice[correct]{$down$}
\choice{$neither$}
\end{multipleChoice}

\end{exercise}






\begin{exercise}
How many $n$-intercepts does the graph of $h(n)$ have?

\begin{multipleChoice}
\choice{$0$}
\choice{$1$}
\choice[correct]{$2$}
\end{multipleChoice}

\end{exercise}
