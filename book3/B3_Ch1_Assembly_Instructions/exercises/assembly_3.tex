\begin{definition}
Let $N$ be some real number.
Let $M$ be some real number.
Let the function $T$ be defined as $T(y) = |y-N|+M$ with a domain of $(-\infty, \infty)$. 

\end{definition}


\begin{exercise}
The least value attained by $T$ is M.

\begin{multipleChoice}
\choice[correct]{True}
\choice{False}
\end{multipleChoice}

\end{exercise}



\begin{exercise}
The greatest value attained by $T$ is M.

\begin{multipleChoice}
\choice{True}
\choice[correct]{False}
\end{multipleChoice}

\end{exercise}







\begin{exercise}
The equation $T(y) = M+1$ has how many solutions?

\begin{selectAll}
\choice{$0$}
\choice{$1$}
\choice[correct]{$2$}
\choice{$3$}
\end{selectAll}

\end{exercise}
