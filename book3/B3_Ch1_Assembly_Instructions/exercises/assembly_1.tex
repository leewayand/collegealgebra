\begin{exercise}
Let the function $B$ be defined as $B(k) = 2(k+3)^2 - 5$ with a domain of $(-\infty, \infty)$. Which of the following are pairs in $B$?

\begin{selectAll}
\choice[correct]{$(-1, 3)$}
\choice[correct]{$(-3, -5)$}
\choice[correct]{$(-2, -3)$}
\choice[correct]{$(0, 13)$}
\end{selectAll}

\end{exercise}
