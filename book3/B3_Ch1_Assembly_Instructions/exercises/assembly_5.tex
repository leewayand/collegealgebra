\begin{definition}
Let the function $f$ be defined as $f(x) = \frac{x+2}{x+3}$ with a domain of $(-\infty, -3) \cup (-3, \infty)$. 

\end{definition}




\begin{exercise}
$f(4) = \answer{6/7} $ 
\end{exercise}





\begin{exercise}
If $A$ is a real number close to -3, but less than -3, then what type of value is $f(A)$?

\begin{multipleChoice}
\choice{Big and Positive}
\choice[correct]{Big and Negative}
\choice{Small and Positive}
\choice{Small and Negative}
\end{multipleChoice}

\end{exercise}
