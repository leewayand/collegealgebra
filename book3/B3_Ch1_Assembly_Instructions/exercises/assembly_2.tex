\outcome{Identify pairs in a relation.}


\begin{document}






\begin{exercise}
Let the function $A$ be defined as $A(g) = |g+2|-3$ with a domain of $(-\infty, \infty)$. Which of the following are pairs in $A$?

\begin{selectAll}
\choice[correct]{$(-1, -2)$}
\choice[correct]{$(-3, -2)$}
\choice[correct]{$(2, 1)$}
\choice[correct]{$(1, 0)$}
\end{selectAll}

\end{exercise}





\begin{exercise}
Let the function $A$ be defined as $A(g) = |g+2|-3$ with a domain of $[0, \infty)$. Which of the following are pairs in $A$?

\begin{selectAll}
\choice{$(-1, -2)$}
\choice{$(-3, -2)$}
\choice[correct]{$(2, 1)$}
\choice[correct]{$(1, 0)$}
\end{selectAll}

\end{exercise}






\begin{exercise}
Let the function $A$ be defined as $A(g) = |g+2|-3$ with a domain of $(-\infty, \infty)$. Then $A(g) > 0$ for all values of $g$.



\begin{multipleChoice}
\choice{True}
\choice[correct]{False}
\end{multipleChoice}

\end{exercise}




\begin{exercise}
Let the function $A$ be defined as $A(g) = |g+2|-3$ with a domain of $(1, \infty)$. Then $A(g) > 0$ for all values of $g$.



\begin{multipleChoice}
\choice[correct]{True}
\choice{False}
\end{multipleChoice}

\end{exercise}
