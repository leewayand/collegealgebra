\begin{definition}
Two piecewise functions are defined below.

$C(y) =   
\left \{ \begin{aligned}     
8 y - 3 && -6 \leq y \leq -3 \\        
-(y-2)^2 + 1 && -3 < y < 4 \\      
\frac{y-5}{y+1} && 4 \leq y < 11\\       
\end{aligned} \right .$ 




$D(x) =   
\left \{ \begin{aligned}     
\frac{x-5}{x-3} && [-6, 3) \\        
\sqrt{x-3} + 1 && [3, 7) \\      
-3|x-3|-2 && [7, 11]\\       
\end{aligned} \right .$ 


\end{definition}





\begin{exercise}
Are the domains of $C$ and $D$ equal?

\begin{multipleChoice}
\choice[correct]{Yes}
\choice{No}
\end{multipleChoice}

\end{exercise}


\begin{exercise}
Is $F(x) = C(x) \cdot D(x)$ well-defined?

\begin{multipleChoice}
\choice[correct]{Yes}
\choice{No}
\end{multipleChoice}

\end{exercise}

\end{exercise}
