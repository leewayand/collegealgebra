\begin{exercise}
Select all reductions of $[-3, 4] \cup (-1, 5) $ 

\begin{selectAll}
\choice{$[-3, 5]$}
\choice{$[-3, 4]$}
\choice[correct]{$[-3, 5)$}
\choice{$(-1, 5]$}
\end{selectAll}

\end{exercise}
