\begin{exercise}
Select all reductions of $[-3, 4] \cup (-1, 5) \cup \{  0 \}$ 

\begin{selectAll}
\choice{$[-3, 5]$}
\choice{$[-3, 5] \cup \{ 0 \} $}
\choice{$\{ 0 \} \cup [-3, 5)$}
\choice[correct]{$[-3, 5)$}
\end{selectAll}

\end{exercise}
