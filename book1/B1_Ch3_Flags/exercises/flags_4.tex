\outcome{Identify pairs in a relation.}



\begin{definition}

Let Bike = \{  A, B, C, D \}.
Let Car = \{  1, 2, 3  \}.

Let Pairs1 = \{ (A, 3), (B, 2), (C, 3), (D, 2)  \}.
Let Pairs2 = \{ (A, 1), (B, 2),  (D, 3)  \}.
Let Pairs3 = \{ (1, C), (2, C), (3, D)  \}.



\end{definition}





\begin{exercise}

Does Pairs1 define Bike as a function of Car?


  \begin{multipleChoice}
    \choice[correct]{Yes}
    \choice{No}
  \end{multipleChoice}

\end{exercise}
